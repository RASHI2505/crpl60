class Solution {
public:
    int findDuplicate(vector<int>& nums) {
        int count=0;
        int n=nums.size();
        sort(nums.begin(), nums.end()) ; //key
        for(int i=0;i<=n;i++)
        {
           if(nums[i]==nums[i+1])
           {
               count=nums[i];
               break;
           }
            
        }
        return count;
        
    }
};
